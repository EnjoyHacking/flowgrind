80 static int name2socket(struct _flow *flow, char *server_name, unsigned port, struct sockaddr **saptr,
81  socklen_t *lenp, char do_connect,
82  const int read_buffer_size_req, int *read_buffer_size,
83  const int send_buffer_size_req, int *send_buffer_size)
84 {
85  int fd, n;
86  struct addrinfo hints, *res, *ressave;
87  struct sockaddr_in *tempv4;
88  struct sockaddr_in6 *tempv6;
89  char service[7];
90 
91  bzero(&hints, sizeof(struct addrinfo));
92  hints.ai_family = AF_UNSPEC;
93  hints.ai_socktype = SOCK_STREAM;
94 
95  snprintf(service, sizeof(service), "%u", port);
96 
97  if ((n = getaddrinfo(server_name, service, &hints, &res)) != 0) {
98  flow_error(flow, "getaddrinfo() failed: %s",
99  gai_strerror(n));
100  return -1;
101  }
102  ressave = res;
103 
104  do {
105  int rc;
106 
107  fd = socket(res->ai_family, res->ai_socktype, res->ai_protocol);
108  if (fd < 0)
109  continue;
110 
111  if (send_buffer_size)
112  *send_buffer_size = set_window_size_directed(fd, send_buffer_size_req, SO_SNDBUF);
113  if (read_buffer_size)
114  *read_buffer_size = set_window_size_directed(fd, read_buffer_size_req, SO_RCVBUF);
115 
116  if (!do_connect)
117  break;
118 
119  rc = connect(fd, res->ai_addr, res->ai_addrlen);
120  if (rc == 0) {
121  if (res->ai_family == PF_INET) {
122  tempv4 = (struct sockaddr_in *) res->ai_addr;
123  strncpy(server_name, inet_ntoa(tempv4->sin_addr), 256);
124  server_name[255] = 0;
125  }
126  else if (res->ai_family == PF_INET6){
127  tempv6 = (struct sockaddr_in6 *) res->ai_addr;
128  inet_ntop(AF_INET6, &tempv6->sin6_addr, server_name, 256);
129  }
130  break;
131  }
132 
133  warn("failed to connect to '%s:%d' ", server_name, port);
134  close(fd);
135  } while ((res = res->ai_next) != NULL);
136 
137  if (res == NULL) {
138  flow_error(flow, "Could not establish connection to "
139  "\"%s:%d\": %s", server_name, port, strerror(errno));
140  freeaddrinfo(ressave);
141  return -1;
142  }
143 
144  if (saptr && lenp) {
145  *saptr = malloc(res->ai_addrlen);
146  if (*saptr == NULL)
147  crit("malloc(): failed");
148  memcpy(*saptr, res->ai_addr, res->ai_addrlen);
149  *lenp = res->ai_addrlen;
150  }
151 
152  freeaddrinfo(ressave);
153 
154  return fd;
155 }

157 int add_flow_source(struct _request_add_flow_source *request)
158 {
159 #ifdef TCP_CONGESTION
160  socklen_t opt_len = 0;
161 #endif /* TCP_CONGESTION */
162  struct _flow *flow;
163 
164  if (num_flows >= MAX_FLOWS) {
165  logging_log(LOG_WARNING, "Can not accept another flow, already handling MAX_FLOW flows.");
166  request_error(&request->r, "Can not accept another flow, already handling MAX_FLOW flows.");
167  return -1;
168  }
169 
170  flow = &flows[num_flows++];
171  init_flow(flow, 1);
172 
173  flow->settings = request->settings;
174  flow->source_settings = request->source_settings;
175  /* be greedy with buffer sizes */
176  flow->write_block = calloc(1, flow->settings.maximum_block_size);
177  flow->read_block = calloc(1, flow->settings.maximum_block_size);
178 
179  if (flow->write_block == NULL || flow->read_block == NULL) {
180  logging_log(LOG_ALERT, "could not allocate memory for read/write blocks");
181  request_error(&request->r, "could not allocate memory for read/write blocks");
182  uninit_flow(flow);
183  num_flows--;
184  return -1;
185  }
186  if (flow->settings.byte_counting) {
187  int byte_idx;
188  for (byte_idx = 0; byte_idx < flow->settings.maximum_block_size; byte_idx++)
189  *(flow->write_block + byte_idx) = (unsigned char)(byte_idx & 0xff);
190  }
191 
192  flow->state = GRIND_WAIT_CONNECT;
193  flow->fd = name2socket(flow, flow->source_settings.destination_host,
194  flow->source_settings.destination_port,
195  &flow->addr, &flow->addr_len, 0,
196  flow->settings.requested_read_buffer_size, &request->real_read_buffer_size,
197  flow->settings.requested_send_buffer_size, &request->real_send_buffer_size);
198  if (flow->fd == -1) {
199  logging_log(LOG_ALERT, "Could not create data socket: %s", flow->error);
200  request_error(&request->r, "Could not create data socket: %s", flow->error);
201  uninit_flow(flow);
202  num_flows--;
203  return -1;
204  }
205 
206  if (set_flow_tcp_options(flow) == -1) {
207  request->r.error = flow->error;
208  flow->error = NULL;
209  uninit_flow(flow);
210  num_flows--;
211  return -1;
212  }
213 
214 #ifdef TCP_CONGESTION
215  opt_len = sizeof(request->cc_alg);
216  if (getsockopt(flow->fd, IPPROTO_TCP, TCP_CONGESTION,
217  request->cc_alg, &opt_len) == -1) {
218  request_error(&request->r, "failed to determine actual congestion control algorithm: %s",
219  strerror(errno));
220  uninit_flow(flow);
221  num_flows--;
222  return -1;
223  }
224 #endif /* TCP_CONGESTION */
225 
226 #ifdef HAVE_LIBPCAP
227  fg_pcap_go(flow);
228 #endif /* HAVE_LIBPCAP */
229  if (!flow->source_settings.late_connect) {
230  DEBUG_MSG(4, "(early) connecting test socket");
231  connect(flow->fd, flow->addr, flow->addr_len);
232  flow->connect_called = 1;
233  flow->pmtu = get_pmtu(flow->fd);
234  }
235 
236  request->flow_id = flow->id;
237 
238  return 0;
239 }
