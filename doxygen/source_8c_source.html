76 static int name2socket(struct flow *flow, char *server_name, unsigned port, struct sockaddr **saptr,
77  socklen_t *lenp, char do_connect,
78  const int read_buffer_size_req, int *read_buffer_size,
79  const int send_buffer_size_req, int *send_buffer_size)
80 {
81  int fd, n;
82  struct addrinfo hints, *res, *ressave;
83  struct sockaddr_in *tempv4;
84  struct sockaddr_in6 *tempv6;
85  char service[7];
86 
87  bzero(&hints, sizeof(struct addrinfo));
88  hints.ai_family = AF_UNSPEC;
89  hints.ai_socktype = SOCK_STREAM;
90 
91  snprintf(service, sizeof(service), "%u", port);
92 
93  if ((n = getaddrinfo(server_name, service, &hints, &res)) != 0) {
94  flow_error(flow, "getaddrinfo() failed: %s",
95  gai_strerror(n));
96  return -1;
97  }
98  ressave = res;
99 
100  do {
101  int rc;
102 
103  fd = socket(res->ai_family, res->ai_socktype, res->ai_protocol);
104  if (fd < 0)
105  continue;
106 
107  if (send_buffer_size)
108  *send_buffer_size = set_window_size_directed(fd, send_buffer_size_req, SO_SNDBUF);
109  if (read_buffer_size)
110  *read_buffer_size = set_window_size_directed(fd, read_buffer_size_req, SO_RCVBUF);
111 
112  if (!do_connect)
113  break;
114 
115  rc = connect(fd, res->ai_addr, res->ai_addrlen);
116  if (rc == 0) {
117  if (res->ai_family == PF_INET) {
118  tempv4 = (struct sockaddr_in *) res->ai_addr;
119  strncpy(server_name, inet_ntoa(tempv4->sin_addr), 256);
120  server_name[255] = 0;
121  }
122  else if (res->ai_family == PF_INET6){
123  tempv6 = (struct sockaddr_in6 *) res->ai_addr;
124  inet_ntop(AF_INET6, &tempv6->sin6_addr, server_name, 256);
125  }
126  break;
127  }
128 
129  warn("failed to connect to '%s:%d' ", server_name, port);
130  close(fd);
131  } while ((res = res->ai_next) != NULL);
132 
133  if (res == NULL) {
134  flow_error(flow, "Could not establish connection to "
135  "\"%s:%d\": %s", server_name, port, strerror(errno));
136  freeaddrinfo(ressave);
137  return -1;
138  }
139 
140  if (saptr && lenp) {
141  *saptr = malloc(res->ai_addrlen);
142  if (*saptr == NULL)
143  crit("malloc(): failed");
144  memcpy(*saptr, res->ai_addr, res->ai_addrlen);
145  *lenp = res->ai_addrlen;
146  }
147 
148  freeaddrinfo(ressave);
149 
150  return fd;
151 }

153 int add_flow_source(struct request_add_flow_source *request)
154 {
155 #ifdef HAVE_SO_TCP_CONGESTION
156  socklen_t opt_len = 0;
157 #endif /* HAVE_SO_TCP_CONGESTION */
158  struct flow *flow;
159 
160  if (fg_list_size(&flows) >= MAX_FLOWS) {
161  logging_log(LOG_WARNING, "Can not accept another flow, already handling MAX_FLOW flows.");
162  request_error(&request->r, "Can not accept another flow, already handling MAX_FLOW flows.");
163  return -1;
164  }
165 
166  flow = malloc(sizeof(struct flow));
167  if (!flow) {
168  logging_log(LOG_ALERT, "could not allocate memory for flow");
169  return -1;
170  }
171 
172  init_flow(flow, 1);
173 
174  flow->settings = request->settings;
175  flow->source_settings = request->source_settings;
176  /* be greedy with buffer sizes */
177  flow->write_block = calloc(1, flow->settings.maximum_block_size);
178  flow->read_block = calloc(1, flow->settings.maximum_block_size);
179 
180  if (flow->write_block == NULL || flow->read_block == NULL) {
181  logging_log(LOG_ALERT, "could not allocate memory for read/write blocks");
182  request_error(&request->r, "could not allocate memory for read/write blocks");
183  uninit_flow(flow);
184  return -1;
185  }
186  if (flow->settings.byte_counting) {
187  int byte_idx;
188  for (byte_idx = 0; byte_idx < flow->settings.maximum_block_size; byte_idx++)
189  *(flow->write_block + byte_idx) = (unsigned char)(byte_idx & 0xff);
190  }
191 
192  flow->state = GRIND_WAIT_CONNECT;
193  flow->fd = name2socket(flow, flow->source_settings.destination_host,
194  flow->source_settings.destination_port,
195  &flow->addr, &flow->addr_len, 0,
196  flow->settings.requested_read_buffer_size, &request->real_read_buffer_size,
197  flow->settings.requested_send_buffer_size, &request->real_send_buffer_size);
198  if (flow->fd == -1) {
199  logging_log(LOG_ALERT, "Could not create data socket: %s", flow->error);
200  request_error(&request->r, "Could not create data socket: %s", flow->error);
201  uninit_flow(flow);
202  return -1;
203  }
204 
205  if (set_flow_tcp_options(flow) == -1) {
206  request->r.error = flow->error;
207  flow->error = NULL;
208  uninit_flow(flow);
209  return -1;
210  }
211 
212 #ifdef HAVE_SO_TCP_CONGESTION
213  opt_len = sizeof(request->cc_alg);
214  if (getsockopt(flow->fd, IPPROTO_TCP, TCP_CONGESTION,
215  request->cc_alg, &opt_len) == -1) {
216  request_error(&request->r, "failed to determine actual congestion control algorithm: %s",
217  strerror(errno));
218  uninit_flow(flow);
219  return -1;
220  }
221 #endif /* HAVE_SO_TCP_CONGESTION */
222 
223 #ifdef HAVE_LIBPCAP
224  fg_pcap_go(flow);
225 #endif /* HAVE_LIBPCAP */
226  if (!flow->source_settings.late_connect) {
227  DEBUG_MSG(4, "(early) connecting test socket");
228  connect(flow->fd, flow->addr, flow->addr_len);
229  flow->connect_called = 1;
230  flow->pmtu = get_pmtu(flow->fd);
231  }
232 
233  request->flow_id = flow->id;
234 
235  fg_list_push_back(&flows, flow);
236 
237  return 0;
238 }
