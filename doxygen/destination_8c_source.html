77 /* listen_port will receive the port of the created socket */
78 static int create_listen_socket(struct _flow *flow, char *bind_addr,
79  unsigned short *listen_port)
80 {
81  int port;
82  int rc;
83  int fd;
84  struct addrinfo hints, *res, *ressave;
85 
86  bzero(&hints, sizeof(struct addrinfo));
87  hints.ai_flags = bind_addr ? 0 : AI_PASSIVE;
88  hints.ai_family = AF_UNSPEC;
89  hints.ai_socktype = SOCK_STREAM;
90 
91  /* Any port will be fine */
92  if ((rc = getaddrinfo(bind_addr, "0", &hints, &res)) != 0) {
93  logging_log(LOG_ALERT, "Error: getaddrinfo() failed: %s\n",
94  gai_strerror(rc));
95  flow_error(flow, "getaddrinfo() failed: %s",
96  gai_strerror(rc));
97  return -1;
98  }
99 
100  ressave = res;
101 
102  do {
103  fd = socket(res->ai_family, res->ai_socktype,
104  res->ai_protocol);
105  if (fd < 0)
106  continue;
107  if (bind(fd, res->ai_addr, res->ai_addrlen) == 0)
108  break;
109  close(fd);
110  } while ((res = res->ai_next) != NULL);
111 
112  if (res == NULL) {
113  logging_log(LOG_ALERT, "failed to create listen socket");
114  flow_error(flow, "Failed to create listen socket: %s",
115  strerror(errno));
116  freeaddrinfo(ressave);
117  return -1;
118  }
119 
120  freeaddrinfo(ressave);
121 
122  /* we need to set sockopt mtcp before we start listen() */
123  if (flow->settings.mtcp)
124  set_tcp_mtcp(fd);
125 
126  if (flow->settings.cc_alg)
127  set_congestion_control(fd, flow->settings.cc_alg);
128 
129  if (listen(fd, 0) < 0) {
130  logging_log(LOG_ALERT, "listen failed: %s",
131  strerror(errno));
132  flow_error(flow, "Listen failed: %s", strerror(errno));
133  return -1;
134  }
135 
136  set_non_blocking(fd);
137 
138  port = get_port(fd);
139  if (port < 0) {
140  flow_error(flow, "Could not get port: %s", strerror(errno));
141  close(fd);
142  return -1;
143  }
144  DEBUG_MSG(LOG_DEBUG, "listening on port %d", port);
145  *listen_port = (unsigned short)port;
146 
147  return fd;
148 }

150 void add_flow_destination(struct _request_add_flow_destination *request)
151 {
152  struct _flow *flow;
153  unsigned short server_data_port;
154 
155  if (num_flows >= MAX_FLOWS) {
156  logging_log(LOG_WARNING, "Can not accept another flow, already "
157  "handling MAX_FLOW flows.");
158  request_error(&request->r, "Can not accept another flow, "
159  "already handling MAX_FLOW flows.");
160  return;
161  }
162 
163  flow = &flows[num_flows++];
164  init_flow(flow, 0);
165 
166  flow->settings = request->settings;
167  flow->write_block = calloc(1, flow->settings.maximum_block_size );
168  flow->read_block = calloc(1, flow->settings.maximum_block_size );
169  if (flow->write_block == NULL || flow->read_block == NULL) {
170  logging_log(LOG_ALERT, "could not allocate memory for "
171  "read/write blocks");
172  request_error(&request->r, "could not allocate memory "
173  "for read/write blocks");
174  uninit_flow(flow);
175  num_flows--;
176  return;
177  }
178 
179  if (flow->settings.byte_counting) {
180  int byte_idx;
181  for (byte_idx = 0; byte_idx < flow->settings.maximum_block_size;
182  byte_idx++)
183  *(flow->write_block + byte_idx) =
184  (unsigned char)(byte_idx & 0xff);
185  }
186 
187  /* Create listen socket for data connection */
188  if ((flow->listenfd_data =
189  create_listen_socket(flow,
190  flow->settings.bind_address[0]
191  ? flow->settings.bind_address : 0,
192  &server_data_port)) == -1) {
193  logging_log(LOG_ALERT, "could not create listen socket for "
194  "data connection: %s", flow->error);
195  request_error(&request->r, "could not create listen socket "
196  "for data connection: %s", flow->error);
197  uninit_flow(flow);
198  num_flows--;
199  return;
200  } else {
201  DEBUG_MSG(LOG_WARNING, "listening on %s port %u for data "
202  "connection", flow->settings.bind_address,
203  server_data_port);
204  }
205 
206  flow->real_listen_send_buffer_size =
207  set_window_size_directed(flow->listenfd_data,
208  flow->settings.requested_send_buffer_size,
209  SO_SNDBUF);
210  flow->real_listen_receive_buffer_size =
211  set_window_size_directed(flow->listenfd_data,
212  flow->settings.requested_read_buffer_size,
213  SO_RCVBUF);
214 
215  request->listen_data_port = (int)server_data_port;
216  request->real_listen_send_buffer_size =
217  flow->real_listen_send_buffer_size;
218  request->real_listen_read_buffer_size =
219  flow->real_listen_receive_buffer_size;
220  request->flow_id = flow->id;
221 
222  return;
223 }

226 {

246  fg_nameinfo((struct sockaddr *)&caddr, addrlen));

257  flow->real_listen_send_buffer_size != real_send_buffer_size) {
258  logging_log(LOG_WARNING, "Failed to set send buffer size of test "

276  real_receive_buffer_size);
277  return -1;
278  }
279  if (set_flow_tcp_options(flow) == -1)
280  return -1;
281  DEBUG_MSG(LOG_NOTICE, "data socket accepted");
282  flow->state = GRIND;
283  flow->connect_called = 1;
284 
285  return 0;
