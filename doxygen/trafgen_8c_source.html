50 inline static double calculate(struct flow *flow, enum distributions type, double param_one, double param_two) {
51 
52  double val = 0;
53 
54  switch (type) {
55  case NORMAL:
56  val = dist_normal (flow, param_one, param_two );
57  DEBUG_MSG(LOG_DEBUG, "calculated normal distribution value %f for parameters %f,%f", val, param_one, param_two);
58  break;
59 
60  case UNIFORM:
61  val = dist_uniform (flow, param_one, param_two );
62  DEBUG_MSG(LOG_DEBUG, "calculated uniform distribution value %f", val);
63  break;
64 
65  case WEIBULL:
66  val = dist_weibull (flow, param_one, param_two );
67  DEBUG_MSG(LOG_DEBUG, "calculated weibull distribution value %f for parameters %f,%f", val, param_one, param_two);
68  break;
69 
70  case EXPONENTIAL:
71  val = dist_exponential (flow, param_one);
72  DEBUG_MSG(LOG_DEBUG, "calculated exponential distribution value %f for parameters %f", val, param_one);
73  break;
74 
75  case PARETO:
76  val = dist_pareto (flow, param_one, param_two);
77  DEBUG_MSG(LOG_DEBUG, "calculated pareto distribution value %f for parameters %f,%f", val, param_one, param_two);
78  break;
79 
80  case LOGNORMAL:
81  val = dist_normal (flow, param_one, param_two );
82  DEBUG_MSG(LOG_DEBUG, "calculated lognormal distribution value %f for parameters %f,%f", val, param_one, param_two);
83  break;
84 
85  case CONSTANT:
86  /* constant is default */
87  default:
88  val = param_one;
89  DEBUG_MSG(LOG_DEBUG, "constant value %f", val);
90 
91  }
92 
93  return val;
94 
95 }
96 int next_request_block_size(struct flow *flow)
97 {
98  int bs = 0;
99  int i = 0;
100  /* recalculate values to match prequisits, but at most 10 times */
101  while (( bs < MIN_BLOCK_SIZE || bs > flow->settings.maximum_block_size) && i < MAX_RUNS_PER_DISTRIBUTION) {
102 
103  bs = round(calculate(
104  flow,
105  flow->settings.request_trafgen_options.distribution,
106  flow->settings.request_trafgen_options.param_one,
107  flow->settings.request_trafgen_options.param_two
108  ));
109  i++;
110  }
111 
112  /* sanity checks */
113  if (i >= MAX_RUNS_PER_DISTRIBUTION && bs < MIN_BLOCK_SIZE) {
114  bs = MIN_BLOCK_SIZE;
115  DEBUG_MSG(LOG_WARNING, "WARNING: applied minimal request size limit %d for flow %d", bs, flow->id);
116  }
117 
118  if (i >= MAX_RUNS_PER_DISTRIBUTION && bs > flow->settings.maximum_block_size) {
119  bs = flow->settings.maximum_block_size;
120  DEBUG_MSG(LOG_WARNING, "WARNING: applied maximal request size limit %d for flow %d", bs, flow->id);
121 
122  }
123 
124  DEBUG_MSG(LOG_NOTICE, "calculated request size %d for flow %d after %d runs", bs, flow->id, i);
125 
126  return bs;
127 }
