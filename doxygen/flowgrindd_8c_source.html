142 static void sighandler(int sig)
143 {
144  int status;
145 
146  switch (sig) {
147  case SIGCHLD:
148  while (waitpid(-1, &status, WNOHANG) > 0)
149  logging_log(LOG_NOTICE, "child returned (status = %d)",
150  status);
151  break;
152 
153  case SIGHUP:
154  logging_log(LOG_NOTICE, "got SIGHUP, don't know what to do.");
155  break;
156 
157  case SIGALRM:
158  logging_log(LOG_NOTICE, "Caught SIGALRM. don't know what to do.");
159  break;
160 
161  case SIGPIPE:
162  break;
163 
164  default:
165  logging_log(LOG_ALERT, "got signal %d, but don't remember "
166  "intercepting it, aborting...", sig);
167  abort();
168  }

171 static int dispatch_request(struct request *request, int type)
172 {
173  pthread_cond_t cond;
174 
175  request->error = NULL;
176  request->type = type;
177  request->next = NULL;
178 
179  /* Create synchronization mutex */
180  if (pthread_cond_init(&cond, NULL)) {
181  request_error(request, "Could not create synchronization mutex");
182  return -1;
183  }
184  request->condition = &cond;
185 
186  pthread_mutex_lock(&mutex);
187 
188  if (!requests) {
189  requests = request;
190  requests_last = request;
191  } else {
192  requests_last->next = request;
193  requests_last = request;
194  }
195  if (write(daemon_pipe[1], &type, 1) != 1) /* Doesn't matter what we write */
196  return -1;
197  /* Wait until the daemon thread has processed the request */
198  pthread_cond_wait(&cond, &mutex);
199 
200  pthread_mutex_unlock(&mutex);
201 
202  if (request->error)
203  return -1;
204 
205  return 0;
206 }

863 static int bind_rpc_server(char *bind_addr, unsigned int port) {
864  int rc;
865  int fd;
866  int optval;
867  struct addrinfo hints, *res, *ressave;
868  char tmp_port[100];
869 
870  bzero(&hints, sizeof(struct addrinfo));
871  hints.ai_flags = AI_PASSIVE | AI_NUMERICSERV;
872  hints.ai_family = AF_UNSPEC;
873  hints.ai_socktype = SOCK_STREAM;
874  sprintf(tmp_port, "%u", port);
875 
876  if ((rc = getaddrinfo(bind_addr, tmp_port,
877  &hints, &res)) != 0) {
878  critx( "Failed to find address to bind rpc_server: %s\n",
879  gai_strerror(rc));
880  return -1;
881  }
882  ressave = res;
883 
884  /* try to bind the first succeeding socket of
885  the returned addresses (libxmlrpc only supports one fd)
886  */
887  do {
888  fd = socket(res->ai_family, res->ai_socktype,
889  res->ai_protocol);
890  if (fd < 0)
891  continue;
892  /* ignore old client connections in TIME_WAIT */
893  optval = 1;
894  setsockopt(fd, SOL_SOCKET, SO_REUSEADDR, &optval, sizeof(optval));
895  /* Disable Nagle algorithm to reduce latency */
896  setsockopt(fd, IPPROTO_TCP, TCP_NODELAY, &optval, sizeof(optval));
897 
898  if (bind(fd, res->ai_addr, res->ai_addrlen) == 0)
899  break;
900 
901  close(fd);
902  } while ((res = res->ai_next) != NULL);
903 
904  if (res == NULL) {
905  crit("failed to bind RPC listen socket");
906  freeaddrinfo(ressave);
907  return -1;
908  }
909 
910  return fd;
911 }
