65 int stack_push(struct _stack *s, const void *elem)
66 {
67  /* Stack is full, we need to realloc() */
68  if (s->used == s->allocated) {
69  s->allocated *= 2;
70  s->elements = realloc(s->elements,
71  s->allocated * s->element_size);
72  if (unlikely(!s->elements))
73  return -1;
74  }
75 
76  void *dst = (void *)s->elements + s->used * s->element_size;
77  memcpy(dst, elem, s->element_size);
78  s->used++;
79 
80  return 0;
81 }

84 {

92 
93  s->used--;
