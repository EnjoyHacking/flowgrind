205 static void prepare_wfds(struct timespec *now, struct _flow *flow, fd_set *wfds)
206 {
207  int rc = 0;
208 
209  if (flow_in_delay(now, flow, WRITE)) {
210  DEBUG_MSG(LOG_WARNING, "flow %i not started yet (delayed)",
211  flow->id);
212  return;
213  }
214 
215  if (flow_sending(now, flow, WRITE)) {
216 #ifdef DEBUG
217  assert(!flow->finished[WRITE]);
218 #endif
219  if (flow_block_scheduled(now, flow)) {
220  DEBUG_MSG(LOG_DEBUG, "adding sock of flow %d to wfds",
221  flow->id);
222  FD_SET(flow->fd, wfds);
223  } else {
224  DEBUG_MSG(LOG_DEBUG, "no block for flow %d scheduled "
225  "yet", flow->id);
226  }
227  } else if (!flow->finished[WRITE]) {
228  flow->finished[WRITE] = 1;
229  if (flow->settings.shutdown) {
230  DEBUG_MSG(LOG_WARNING, "shutting down flow %d (WR)",
231  flow->id);
232  rc = shutdown(flow->fd,SHUT_WR);
233  if (rc == -1)
234  error(ERR_WARNING, "shutdown() SHUT_WR failed: "
235  "%s", strerror(errno));
236  }
237  }
238 
239  return;
240 }

242 static int prepare_rfds(struct timespec *now, struct _flow *flow, fd_set *rfds)
243 {
244  int rc = 0;
245 
246  if (!flow_in_delay(now, flow, READ) && !flow_sending(now, flow, READ)) {
247  if (!flow->finished[READ] && flow->settings.shutdown) {
248  error(ERR_WARNING, "server flow %u missed to shutdown",
249  flow->id);
250  rc = shutdown(flow->fd, SHUT_RD);
251  if (rc == -1)
252  error(ERR_WARNING, "shutdown SHUT_RD "
253  "failed: %s", strerror(errno));
254  flow->finished[READ] = 1;
255  }
256  }
257 
258  if (flow->source_settings.late_connect && !flow->connect_called ) {
259  DEBUG_MSG(LOG_ERR, "late connecting test socket for flow %d "
260  "after %.3fs delay",
261  flow->id, flow->settings.delay[WRITE]);
262  rc = connect(flow->fd, flow->addr, flow->addr_len);
263  if (rc == -1 && errno != EINPROGRESS) {
264  flow_error(flow, "Connect failed: %s", strerror(errno));
265  return -1;
266  }
267  flow->connect_called = 1;
268  flow->pmtu = get_pmtu(flow->fd);
269  }
270 
271  /* Altough the server flow might be finished we keep the socket in
272  * rfd in order to check for buggy servers */
273  if (flow->connect_called && !flow->finished[READ]) {
274  DEBUG_MSG(LOG_DEBUG, "adding sock of flow %d to rfds",
275  flow->id);
276  FD_SET(flow->fd, rfds);
277  }
278 
279  return 0;
280 }

282 static int prepare_fds() {
283 
284  DEBUG_MSG(LOG_DEBUG, "prepare_fds() called, num_flows: %d", num_flows);
285  unsigned int i = 0;
286 
287  FD_ZERO(&rfds);
288  FD_ZERO(&wfds);
289  FD_ZERO(&efds);
290 
291  FD_SET(daemon_pipe[0], &rfds);
292  maxfd = daemon_pipe[0];
293 
294  struct timespec now;
295  gettime(&now);
296 
297  while (i < num_flows) {
298  struct _flow *flow = &flows[i++];
299 
300  if (started &&
301  (flow->finished[READ] ||
302  !flow->settings.duration[READ] ||
303  (!flow_in_delay(&now, flow, READ) &&
304  !flow_sending(&now, flow, READ))) &&
305  (flow->finished[WRITE] ||
306  !flow->settings.duration[WRITE] ||
307  (!flow_in_delay(&now, flow, WRITE) &&
308  !flow_sending(&now, flow, WRITE)))) {
309 
310  /* On Other OSes than Linux or FreeBSD, tcp_info will contain all zeroes */
311  flow->statistics[FINAL].has_tcp_info =
312  get_tcp_info(flow,
313  &flow->statistics[FINAL].tcp_info)
314  ? 0 : 1;
315 
316  flow->pmtu = get_pmtu(flow->fd);
317 
318  if (flow->settings.reporting_interval)
319  report_flow(flow, INTERVAL);
320  report_flow(flow, FINAL);
321  uninit_flow(flow);
322  remove_flow(--i);
323  continue;
324  }
325 
326  if (flow->state == GRIND_WAIT_ACCEPT &&
327  flow->listenfd_data != -1) {
328  FD_SET(flow->listenfd_data, &rfds);
329  maxfd = MAX(maxfd, flow->listenfd_data);
330  }
331 
332  if (!started)
333  continue;
334 
335  if (flow->fd != -1) {
336  FD_SET(flow->fd, &efds);
337  maxfd = MAX(maxfd, flow->fd);
338  prepare_wfds(&now, flow, &wfds);
339  prepare_rfds(&now, flow, &rfds);
340  }
341  }
342 
343  return num_flows;
344 }

446 static void process_requests()
447 {
448  int rc;
449  DEBUG_MSG(LOG_DEBUG, "process_requests trying to lock mutex");
450  pthread_mutex_lock(&mutex);
451  DEBUG_MSG(LOG_DEBUG, "process_requests locked mutex");
452 
453  char tmp[100];
454  for (;;) {
455  int rc = read(daemon_pipe[0], tmp, 100);
456  if (rc != 100)
457  break;
458  }
459 
460  while (requests) {
461  struct _request* request = requests;
462  requests = requests->next;
463  rc = 0;
464 
465  switch (request->type) {
466  case REQUEST_ADD_DESTINATION:
467  add_flow_destination((struct
468  _request_add_flow_destination
469  *)request);
470  break;
471  case REQUEST_ADD_SOURCE:
472  rc = add_flow_source((struct
473  _request_add_flow_source
474  *)request);
475  break;
476  case REQUEST_START_FLOWS:
477  start_flows((struct _request_start_flows *)request);
478  break;
479  case REQUEST_STOP_FLOW:
480  stop_flow((struct _request_stop_flow *)request);
481  break;
482  case REQUEST_GET_STATUS:
483  {
484  struct _request_get_status *r =
485  (struct _request_get_status *)request;
486  r->started = started;
487  r->num_flows = num_flows;
488  }
489  break;
490  default:
491  request_error(request, "Unknown request type");
492  break;
493  }
494  if (rc != 1)
495  pthread_cond_signal(request->condition);
496  }
497 
498  pthread_mutex_unlock(&mutex);
499  DEBUG_MSG(LOG_DEBUG, "process_requests unlocked mutex");
500 }

502 /*
503  * Prepare a report. type is either INTERVAL or FINAL
504  */
505 static void report_flow(struct _flow* flow, int type)
506 {
507  DEBUG_MSG(LOG_DEBUG, "report_flow called for flow %d (type %d)",
508  flow->id, type);
509  struct _report* report =
510  (struct _report*)malloc(sizeof(struct _report));
511 
512  report->id = flow->id;
513  report->type = type;
514 
515  if (type == INTERVAL)
516  report->begin = flow->last_report_time;
517  else
518  report->begin = flow->first_report_time;
519 
520  gettime(&report->end);
521  flow->last_report_time = report->end;
522 
523  /* abort if we were scheduled way to early for a interval report */
524  if (time_diff(&report->begin,&report->end) < 0.2 *
525  flow->settings.reporting_interval && type == INTERVAL){
526  free(report);
527  return;
528  }
529 
530  report->bytes_read = flow->statistics[type].bytes_read;
531  report->bytes_written = flow->statistics[type].bytes_written;
532  report->request_blocks_read =
533  flow->statistics[type].request_blocks_read;
534  report->response_blocks_read =
535  flow->statistics[type].response_blocks_read;
536  report->request_blocks_written =
537  flow->statistics[type].request_blocks_written;
538  report->response_blocks_written =
539  flow->statistics[type].response_blocks_written;
540 
541  report->rtt_min = flow->statistics[type].rtt_min;
542  report->rtt_max = flow->statistics[type].rtt_max;
543  report->rtt_sum = flow->statistics[type].rtt_sum;
544  report->iat_min = flow->statistics[type].iat_min;
545  report->iat_max = flow->statistics[type].iat_max;
546  report->iat_sum = flow->statistics[type].iat_sum;
547  report->delay_min = flow->statistics[type].delay_min;
548  report->delay_max = flow->statistics[type].delay_max;
549  report->delay_sum = flow->statistics[type].delay_sum;
550 
551  /* Currently this will only contain useful information on Linux
552  * and FreeBSD */
553  report->tcp_info = flow->statistics[type].tcp_info;
554 
555  if (flow->fd != -1) {
556  /* Get latest MTU */
557  flow->pmtu = get_pmtu(flow->fd);
558  report->pmtu = flow->pmtu;
559  if (type == FINAL)
560  report->imtu = get_imtu(flow->fd);
561  else
562  report->imtu = 0;
563  } else {
564  report->imtu = 0;
565  report->pmtu = 0;
566  }
567  /* Add status flags to report */
568  report->status = 0;
569 
570  if (flow->statistics[type].bytes_read == 0) {
571  if (flow_in_delay(&report->end, flow, READ))
572  report->status |= 'd';
573  else if (flow_sending(&report->end, flow, READ))
574  report->status |= 'l';
575  else if (flow->settings.duration[READ] == 0)
576  report->status |= 'o';
577  else
578  report->status |= 'f';
579  } else {
580  if (!flow_sending(&report->end, flow, READ) && !flow->finished)
581  report->status |= 'c';
582  else
583  report->status |= 'n';
584  }
585  report->status <<= 8;
586 
587  if (flow->statistics[type].bytes_written == 0) {
588  if (flow_in_delay(&report->end, flow, WRITE))
589  report->status |= 'd';
590  else if (flow_sending(&report->end, flow, WRITE))
591  report->status |= 'l';
592  else if (flow->settings.duration[WRITE] == 0)
593  report->status |= 'o';
594  else
595  report->status |= 'f';
596  } else {
597  if (!flow_sending(&report->end, flow, WRITE) && !flow->finished)
598  report->status |= 'c';
599  else
600  report->status |= 'n';
601  }
602 
603  /* New report interval, reset old data */
604  if (type == INTERVAL) {
605  flow->statistics[INTERVAL].bytes_read = 0;
606  flow->statistics[INTERVAL].bytes_written = 0;
607 
608  flow->statistics[INTERVAL].request_blocks_read = 0;
609  flow->statistics[INTERVAL].response_blocks_read = 0;
610 
611  flow->statistics[INTERVAL].request_blocks_written = 0;
612  flow->statistics[INTERVAL].response_blocks_written = 0;
613 
614  flow->statistics[INTERVAL].rtt_min = FLT_MAX;
615  flow->statistics[INTERVAL].rtt_max = FLT_MIN;
616  flow->statistics[INTERVAL].rtt_sum = 0.0F;
617  flow->statistics[INTERVAL].iat_min = FLT_MAX;
618  flow->statistics[INTERVAL].iat_max = FLT_MIN;
619  flow->statistics[INTERVAL].iat_sum = 0.0F;
620  flow->statistics[INTERVAL].delay_min = FLT_MAX;
621  flow->statistics[INTERVAL].delay_max = FLT_MIN;
622  flow->statistics[INTERVAL].delay_sum = 0.0F;
623  }
624 
625  add_report(report);
626  DEBUG_MSG(LOG_DEBUG, "report_flow finished for flow %d (type %d)",
627  flow->id, type);
628 }

700  flow->settings.reporting_interval);
701  } while (time_is_after(&now, &flow->next_report_time));
702  }

706 static void process_select(fd_set *rfds, fd_set *wfds, fd_set *efds)
707 {
708  unsigned int i = 0;
709  while (i < num_flows) {
710 
711  struct _flow *flow = &flows[i];
712 
713  DEBUG_MSG(LOG_DEBUG, "processing pselect() for flow %d",
714  flow->id);
715 
716  if (flow->listenfd_data != -1 &&
717  FD_ISSET(flow->listenfd_data, rfds)) {
718  DEBUG_MSG(LOG_DEBUG, "ready for accept");
719  if (flow->state == GRIND_WAIT_ACCEPT) {
720  if (accept_data(flow) == -1) {
721  DEBUG_MSG(LOG_ERR, "accept_data() "
722  "failed");
723  goto remove;
724  }
725  }
726  }
727 
728  if (flow->fd != -1) {
729  if (FD_ISSET(flow->fd, efds)) {
730  int error_number, rc;
731  socklen_t error_number_size =
732  sizeof(error_number);
733  DEBUG_MSG(LOG_DEBUG, "sock of flow %d in efds",
734  flow->id);
735  rc = getsockopt(flow->fd, SOL_SOCKET,
736  SO_ERROR,
737  (void *)&error_number,
738  &error_number_size);
739  if (rc == -1) {
740  error(ERR_WARNING, "failed to get "
741  "errno for non-blocking connect: "
742  "%s", strerror(errno));
743  goto remove;
744  }
745  if (error_number != 0) {
746  fprintf(stderr, "connect: %s\n",
747  strerror(error_number));
748  goto remove;
749  }
750  }
751  if (FD_ISSET(flow->fd, wfds))
752  if (write_data(flow) == -1) {
753  DEBUG_MSG(LOG_ERR, "write_data() failed");
754  goto remove;
755  }
756 
757  if (FD_ISSET(flow->fd, rfds))
758  if (read_data(flow) == -1) {
759  DEBUG_MSG(LOG_ERR, "read_data() failed");
760  goto remove;
761  }
762  }
763  i++;
764  continue;
765 remove:
766  if (flow->fd != -1) {
767  flow->statistics[FINAL].has_tcp_info =
768  get_tcp_info(flow,
769  &flow->statistics[FINAL].tcp_info)
770  ? 0 : 1;
771  }
772  flow->pmtu = get_pmtu(flow->fd);
773  report_flow(flow, FINAL);
774  uninit_flow(flow);
775  remove_flow(i);
776  DEBUG_MSG(LOG_ERR, "removed flow %d", flow->id);
777  }
778 }

908 static int write_data(struct _flow *flow)
909 {
910  int rc = 0;
911  int response_block_size = 0;
912  double interpacket_gap = .0;
913  for (;;) {
914 
915  /* fill buffer with new data */
916  if (flow->current_block_bytes_written == 0) {
917  flow->current_write_block_size =
918  next_request_block_size(flow);
919  response_block_size = next_response_block_size(flow);
920  /* serialize data:
921  * this_block_size */
922  ((struct _block *)flow->write_block)->this_block_size =
923  htonl(flow->current_write_block_size);
924  /* requested_block_size */
925  ((struct _block *)flow->write_block)->request_block_size =
926  htonl(response_block_size);
927  /* write rtt data (will be echoed back by the receiver
928  * in the response packet) */
929  gettime((struct timespec *)
930  (flow->write_block + 2 * (sizeof (int32_t))));
931 
932  DEBUG_MSG(LOG_DEBUG, "wrote new request data to out "
933  "buffer bs = %d, rqs = %d, on flow %d",
934  ntohl(((struct _block *)flow->write_block)->this_block_size),
935  ntohl(((struct _block *)flow->write_block)->request_block_size),
936  flow->id);
937  }
938 
939  rc = write(flow->fd,
940  flow->write_block +
941  flow->current_block_bytes_written,
942  flow->current_write_block_size -
943  flow->current_block_bytes_written);
944 
945  if (rc == -1) {
946  if (errno == EAGAIN) {
947  logging_log(LOG_WARNING, "write queue limit hit "
948  "for flow %d", flow->id);
949  break;
950  }
951  DEBUG_MSG(LOG_WARNING, "write() returned %d on flow %d, "
952  "fd %d: %s", rc, flow->id, flow->fd,
953  strerror(errno));
954  flow_error(flow, "Premature end of test: %s",
955  strerror(errno));
956  return rc;
957  }
958 
959  if (rc == 0) {
960  DEBUG_MSG(LOG_CRIT, "flow %d sent zero bytes. what "
961  "does that mean?", flow->id);
962  return rc;
963  }
964 
965  DEBUG_MSG(LOG_DEBUG, "flow %d sent %d request bytes of %u "
966  "(before = %u)", flow->id, rc,
967  flow->current_write_block_size,
968  flow->current_block_bytes_written);
969 
970  for (int i = 0; i < 2; i++)
971  flow->statistics[i].bytes_written += rc;
972 
973  flow->current_block_bytes_written += rc;
974 
975  if (flow->current_block_bytes_written >=
976  flow->current_write_block_size) {
977 #ifdef DEBUG
978  assert(flow->current_block_bytes_written ==
979  flow->current_write_block_size);
980 #endif
981  /* we just finished writing a block */
982  flow->current_block_bytes_written = 0;
983  gettime(&flow->last_block_written);
984  for (int i = 0; i < 2; i++)
985  flow->statistics[i].request_blocks_written++;
986 
987  interpacket_gap = next_interpacket_gap(flow);
988 
989  /* if we calculated a non-zero packet add relative time
990  * to the next write stamp which is then checked in the
991  * select call */
992  if (interpacket_gap) {
993  time_add(&flow->next_write_block_timestamp,
994  interpacket_gap);
995  if (time_is_after(&flow->last_block_written,
996  &flow->next_write_block_timestamp)) {
997  DEBUG_MSG(LOG_WARNING, "incipient "
998  "congestion on flow %u new "
999  "block scheduled for %s, "
1000  "%.6lfs before now.",
1001  flow->id,
1002  ctimespec(&flow->next_write_block_timestamp),
1003  time_diff(&flow->next_write_block_timestamp,
1004  &flow->last_block_written));
1005  flow->congestion_counter++;
1006  if (flow->congestion_counter >
1007  CONGESTION_LIMIT &&
1008  flow->settings.flow_control)
1009  return -1;
1010  }
1011  }
1012  if (flow->settings.cork && toggle_tcp_cork(flow->fd) == -1)
1013  DEBUG_MSG(LOG_NOTICE, "failed to recork test "
1014  "socket for flow %d: %s",
1015  flow->id, strerror(errno));
1016  }
1017 
1018  if (!flow->settings.pushy)
1019  break;
1020  }
1021  return 0;
1022 }

1024 static inline int try_read_n_bytes(struct _flow *flow, int bytes)
1025 {
1026  int rc;
1027  struct iovec iov;
1028  struct msghdr msg;
1029 /* we only read out of band data for debugging purpose */
1030 #ifdef DEBUG
1031  char cbuf[512];
1032  struct cmsghdr *cmsg;
1033 #else
1034  char cbuf[16];
1035 #endif
1036  iov.iov_base = flow->read_block +
1037  flow->current_block_bytes_read;
1038  iov.iov_len = bytes;
1039  /* no name required */
1040  msg.msg_name = NULL;
1041  msg.msg_namelen = 0;
1042  msg.msg_iov = &iov;
1043  msg.msg_iovlen = 1;
1044  msg.msg_control = cbuf;
1045  msg.msg_controllen = sizeof(cbuf);
1046 
1047  rc = recvmsg(flow->fd, &msg, 0);
1048 
1049  DEBUG_MSG(LOG_DEBUG, "tried reading %d bytes, got %d", bytes, rc);
1050 
1051  if (rc == -1) {
1052  if (errno == EAGAIN)
1053  flow_error(flow, "Premature end of test: %s",
1054  strerror(errno));
1055  return -1;
1056  }
1057 
1058  if (rc == 0) {
1059  DEBUG_MSG(LOG_ERR, "server shut down test socket of "
1060  "flow %d", flow->id);
1061  if (!flow->finished[READ] || !flow->settings.shutdown)
1062  error(ERR_WARNING, "Premature shutdown of server "
1063  "flow");
1064  flow->finished[READ] = 1;
1065  return -1;
1066  }
1067 
1068  DEBUG_MSG(LOG_DEBUG, "flow %d received %u bytes", flow->id, rc);
1069 
1070  flow->current_block_bytes_read += rc;
1071  for (int i = 0; i < 2; i++)
1072  flow->statistics[i].bytes_read += rc;
1073 
1074 #ifdef DEBUG
1075  for (cmsg = CMSG_FIRSTHDR(&msg); cmsg; cmsg = CMSG_NXTHDR(&msg, cmsg))
1076  DEBUG_MSG(LOG_NOTICE, "flow %d received cmsg: type = %u, "
1077  "len = %zu",
1078  flow->id, cmsg->cmsg_type, cmsg->cmsg_len);
1079 #endif
1080 
1081  return rc;
1082 }

1084 static int read_data(struct _flow *flow)
1085 {
1086  int rc = 0;
1087  int optint = 0;
1088  int requested_response_block_size = 0;
1089 
1090  for (;;) {
1091  /* make sure to read block header for new block */
1092  if (flow->current_block_bytes_read < MIN_BLOCK_SIZE) {
1093  rc = try_read_n_bytes(flow,
1094  MIN_BLOCK_SIZE-flow->current_block_bytes_read);
1095  if (flow->current_block_bytes_read < MIN_BLOCK_SIZE)
1096  break;
1097  }
1098  /* parse data and update status */
1099 
1100  /* parse and check current block size for validity */
1101  optint = ntohl( ((struct _block *)flow->read_block)->this_block_size );
1102  if (optint >= MIN_BLOCK_SIZE &&
1103  optint <= flow->settings.maximum_block_size )
1104  flow->current_read_block_size = optint;
1105  else
1106  logging_log(LOG_WARNING, "flow %d parsed illegal cbs %d, "
1107  "ignoring (max: %d)", flow->id, optint,
1108  flow->settings.maximum_block_size);
1109 
1110  /* parse and check current request size for validity */
1111  optint = ntohl( ((struct _block *)flow->read_block)->request_block_size );
1112  if (optint == -1 || optint == 0 ||
1113  (optint >= MIN_BLOCK_SIZE &&
1114  optint <= flow->settings.maximum_block_size))
1115  requested_response_block_size = optint;
1116  else
1117  logging_log(LOG_WARNING, "flow %d parsed illegal qbs "
1118  "%d, ignoring (max: %d)",
1119  flow->id,
1120  optint,
1121  flow->settings.maximum_block_size);
1122 #ifdef DEBUG
1123  if (requested_response_block_size == -1) {
1124  DEBUG_MSG(LOG_NOTICE, "processing response block on "
1125  "flow %d size: %d", flow->id,
1126  flow->current_read_block_size);
1127  } else {
1128  DEBUG_MSG(LOG_NOTICE, "processing request block on "
1129  "flow %d size: %d, request: %d",
1130  flow->id,
1131  flow->current_read_block_size,
1132  requested_response_block_size);
1133  }
1134 #endif
1135  /* read rest of block, if we have more to read */
1136  if (flow->current_block_bytes_read <
1137  flow->current_read_block_size)
1138  rc += try_read_n_bytes(flow,
1139  flow->current_read_block_size -
1140  flow->current_block_bytes_read);
1141 
1142  if (flow->current_block_bytes_read >=
1143  flow->current_read_block_size ) {
1144 #ifdef DEBUG
1145  assert(flow->current_block_bytes_read ==
1146  flow->current_read_block_size);
1147 #endif
1148  flow->current_block_bytes_read = 0;
1149 
1150  /* TODO process_rtt(), process_iat(), and
1151  * process_delay () call all gettime().
1152  * Quite inefficient... */
1153 
1154  if (requested_response_block_size == -1) {
1155  /* this is a response block, consider DATA as
1156  * RTT */
1157  for (int i = 0; i < 2; i++)
1158  flow->statistics[i].response_blocks_read++;
1159  process_rtt(flow);
1160  } else {
1161  /* this is a request block, calculate IAT */
1162  for (int i = 0; i < 2; i++)
1163  flow->statistics[i].request_blocks_read++;
1164  process_iat(flow);
1165  process_delay(flow);
1166 
1167  /* send response if requested */
1168  if (requested_response_block_size >=
1169  (signed)MIN_BLOCK_SIZE && !flow->finished[READ])
1170  send_response(flow,
1171  requested_response_block_size);
1172  }
1173  }
1174  if (!flow->settings.pushy)
1175  break;
1176  }
1177  return rc;
1178 }

1275 static void send_response(struct _flow* flow, int requested_response_block_size)
1276 {
1277  int rc;
1278  int try = 0;
1279 #ifdef DEBUG
1280  assert(!flow->current_block_bytes_written);
1281 #endif
1282  /* write requested block size as current size */
1283  ((struct _block *)flow->write_block)->this_block_size =
1284  htonl(requested_response_block_size);
1285  /* rqs = -1 indicates response block */
1286  ((struct _block *)flow->write_block)->request_block_size = htonl(-1);
1287  /* copy rtt data from received block to response block (echo back) */
1288  ((struct _block *)flow->write_block)->data =
1289  ((struct _block *)flow->read_block)->data;
1290  /* workaround for 64bit sender and 32bit receiver: we check if the
1291  * timespec is 64bit and then echo the missing 32bit back, too */
1292  if ((((struct _block *)flow->write_block)->data.tv_sec) ||
1293  ((struct _block *)flow->write_block)->data.tv_nsec)
1294  ((struct _block *)flow->write_block)->data2 =
1295  ((struct _block *)flow->read_block)->data2;
1296 
1297  DEBUG_MSG(LOG_DEBUG, "wrote new response data to out buffer bs = %d, "
1298  "rqs = %d on flow %d",
1299  ntohl(((struct _block *)flow->write_block)->this_block_size),
1300  ntohl(((struct _block *)flow->write_block)->request_block_size),
1301  flow->id);
1302 
1303  /* send data out until block is finished (or abort if 0 zero bytes are
1304  * send CONGESTION_LIMIT times) */
1305  for (;;) {
1306  rc = write(flow->fd,
1307  flow->write_block + flow->current_block_bytes_written,
1308  requested_response_block_size -
1309  flow->current_block_bytes_written);
1310 
1311  DEBUG_MSG(LOG_NOTICE, "send %d bytes response (rqs %d) on flow "
1312  "%d", rc, requested_response_block_size,flow->id);
1313 
1314  if (rc == -1) {
1315  if (errno == EAGAIN) {
1316  DEBUG_MSG(LOG_DEBUG,
1317  "%s, still trying to send response "
1318  "block (write queue hit limit)",
1319  strerror(errno));
1320  try++;
1321  if (try >= CONGESTION_LIMIT &&
